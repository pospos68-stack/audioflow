<script>

/* ===== SAFE START ===== */
console.log("AudioFlow start");

/* data */
const categories=["Román","Detektivka","Sci-fi","Fantasy","Osobní rozvoj","Historie","Thriller","Ostatní"];
let db=null;
let books=[];
let selectedCategory="Vše";

let currentUrl=null;

/* elements */
const audio=document.getElementById("audio");
const library=document.getElementById("library");
const progressBar=document.getElementById("progressBar");
const menu=document.getElementById("menu");
const slider=document.getElementById("gridSlider");
const searchInput=document.getElementById("searchInput");
const nowPlaying=document.getElementById("nowPlaying");

/* slider */
slider.value=localStorage.getItem("cols")||2;
document.documentElement.style.setProperty("--cols",slider.value);

slider.oninput=()=>{
localStorage.setItem("cols",slider.value);
document.documentElement.style.setProperty("--cols",slider.value);
render();
};

/* tabs */
function renderTabs(){
const tabs=document.getElementById("categoryTabs");
tabs.innerHTML="";

["⏱ Pokračovat","🕒 Naposled poslouchané","🆕 Naposled přidané","⭐ Oblíbené","Vše",...categories]
.forEach(cat=>{
const d=document.createElement("div");
d.className="tab"+(cat===selectedCategory?" active":"");
d.innerText=cat;
d.onclick=()=>{
selectedCategory=cat;
renderTabs();
render();
};
tabs.appendChild(d);
});
}

/* DB */
try{
const req=indexedDB.open("AudioBooksDB",1);

req.onupgradeneeded=e=>{
db=e.target.result;
db.createObjectStore("books",{keyPath:"name"});
};

req.onsuccess=e=>{
db=e.target.result;
loadBooks();
};

req.onerror=e=>{
console.error("DB error",e);
};

}catch(e){
console.error("IndexedDB fail",e);
}

/* save */
function saveBook(b){
if(!db) return;
db.transaction("books","readwrite").objectStore("books").put(b);
}

/* load */
function loadBooks(){
db.transaction("books","readonly").objectStore("books").getAll().onsuccess=e=>{
books=e.target.result || [];
renderTabs();
render();
};
}

/* upload */
document.getElementById("fileInput").onchange=async function(){

let grouped={};

for(let f of this.files){

let folder=f.name.split("/")[0];

if(!grouped[folder]){
grouped[folder]={
name:folder,
chapters:[],
cover:null,
playlist:null,
currentChapter:0,
currentTime:0,
category:"Ostatní",
favorite:false,
createdAt:Date.now(),
lastPlayed:0
};
}

/* playlist */
if(f.name.endsWith(".m3u")){
grouped[folder].playlist = await f.text();
continue;
}

/* audio */
if(f.type.startsWith("audio")){
grouped[folder].chapters.push({
name:f.name,
data:await f.arrayBuffer()
});
}

/* cover */
if(f.type.startsWith("image")){
grouped[folder].cover=await f.arrayBuffer();
}
}

/* sort */
Object.values(grouped).forEach(b=>{

if(b.playlist){
let order=b.playlist
.split("\n")
.map(l=>l.trim())
.filter(l=>l && !l.startsWith("#"))
.map(l=>l.split("/").pop());

b.chapters.sort((a,b)=>order.indexOf(a.name)-order.indexOf(b.name));
}else{
b.chapters.sort((a,b)=>a.name.localeCompare(b.name,undefined,{numeric:true}));
}

saveBook(b);
});

loadBooks();
};

/* search */
searchInput.oninput=()=>render();

/* menu */
function openMenu(x,y,book){
menu.innerHTML="";
menu.style.display="flex";
menu.style.left=x+"px";
menu.style.top=y+"px";

/* categories */
categories.forEach(cat=>{
let d=document.createElement("div");
d.innerText="📂 "+cat;
d.onclick=()=>{
book.category=cat;
saveBook(book);
menu.style.display="none";
render();
};
menu.appendChild(d);
});

/* reset */
let r=document.createElement("div");
r.innerText="🔄 Reset přehrávání";
r.onclick=()=>{
book.currentTime=0;
book.currentChapter=0;
saveBook(book);
menu.style.display="none";
};
menu.appendChild(r);
}

/* close menu */
window.onclick=(e)=>{
if(!menu.contains(e.target)){
menu.style.display="none";
}
};

/* render */
function render(){
library.innerHTML="";
let list=[...books];
let q=searchInput.value.toLowerCase();

/* sort */
if(selectedCategory==="🆕 Naposled přidané"){
list.sort((a,b)=>b.createdAt-a.createdAt);
}
if(selectedCategory==="🕒 Naposled poslouchané"){
list.sort((a,b)=>b.lastPlayed-a.lastPlayed);
}
if(selectedCategory==="⏱ Pokračovat"){
list=list.filter(b=>b.currentTime>10 || b.currentChapter>0);
}

/* filter */
list=list.filter(b=>{
if(selectedCategory==="⭐ Oblíbené"&&!b.favorite)return false;
if(categories.includes(selectedCategory)&&b.category!==selectedCategory)return false;
if(q&&!b.name.toLowerCase().includes(q))return false;
return true;
});

/* render cards */
list.forEach(b=>{
let div=document.createElement("div");
div.className="book";

let cover=b.cover?URL.createObjectURL(new Blob([b.cover])):"";

div.innerHTML=`
${cover?`<img src="${cover}">`:`<div style="width:100%;height:100%;background:#333"></div>`}
<div class="book-cat">📂 ${b.category}</div>
<div class="book-title">${b.name}</div>
<div class="fav">${b.favorite?"⭐":"☆"}</div>
`;

div.onclick=()=>play(b);

div.querySelector(".fav").onclick=(e)=>{
e.stopPropagation();
b.favorite=!b.favorite;
saveBook(b);
render();
};

div.oncontextmenu=(e)=>{
e.preventDefault();
openMenu(e.pageX,e.pageY,b);
};

library.appendChild(div);
});
}

/* play */
function play(b){

b.lastPlayed=Date.now();

if(currentUrl) URL.revokeObjectURL(currentUrl);

let ch=b.currentChapter||0;
let t=b.currentTime||0;
let chapter=b.chapters[ch];

currentUrl=URL.createObjectURL(new Blob([chapter.data]));

audio.src=currentUrl;
audio.currentTime=t;
audio.play();

nowPlaying.innerText=`🎧 ${b.name} — ${chapter.name}`;

audio.ontimeupdate=()=>{
b.currentTime=audio.currentTime;

if(audio.duration){
progressBar.style.width=(audio.currentTime/audio.duration*100)+"%";
}

saveBook(b);
};

audio.onended=()=>{
if(b.currentChapter+1 < b.chapters.length){
b.currentChapter++;
b.currentTime=0;
play(b);
}
};
}

/* skip */
function skip(s){
audio.currentTime+=s;
}
